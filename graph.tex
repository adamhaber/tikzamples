  \path (0,0) node (v0) {$v_0$};
  \path (0*72:1cm) node (v1) {$v_1$};
  \path (1*72:1cm) node (v2) {$v_2$};
  \path (2*72:1cm) node (v3) {$v_3$};

  \path (4*72:1cm) node (v5) {$v_5$};

  \draw (v0) -- (v1)
        (v0) -- (v2)
        (v0) -- (v3)
        (v0) -- (v4)
        (v0) -- (v5);
