\documentclass{article}

\usepackage{tikz}


\begin{document}

\begin{tikzpicture}
  \draw [step=0.25cm,color=gray] (-1, -1) grid (1, 1);

  \draw (0, 1) -- (1, 0);
  \draw (0, 1) -- (1, 0) -- (0, -1);
  \draw (0, 1) -- (1, 0) -- (0, -1) -- (-1, 0);
  \draw (0, 1) -- (1, 0) -- (0, -1) -- (-1, 0) -- cycle;
\end{tikzpicture}


\end{document}
